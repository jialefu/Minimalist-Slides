        \centering
        \vspace*{7mm}

            {\color{themecolor}

            \insertsectionnumber.\ \insertsection}\\

                \fill[black!30] (0,0) rectangle (#1\textwidth, 1pt);
                \fill[themecolor] (0,0) rectangle (#1\textwidth/2, 1pt);
            \end{tikzpicture}
        \end{tabular}
    \end{frame}
}
